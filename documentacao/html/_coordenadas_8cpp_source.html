90 double Coordenadas::calcDistancia(const Coordenadas &coordenadas) const
91 {
92  double earthRadius = 3958.75;
93  double dLat = toRad(coordenadas.latitude - latitude);
94  double dLng = toRad(coordenadas.longitude - longitude);
95  double a = sin(dLat / 2) * sin(dLat / 2) +
96  cos(toRad(latitude)) * cos(toRad(coordenadas.latitude)) *
97  sin(dLng / 2) * sin(dLng / 2);
98  double c = 2 * atan2(sqrt(a), sqrt(1 - a));
